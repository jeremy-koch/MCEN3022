Flat Plate — Interactive Demo

  • Show All Code
  • Hide All Code
  • View Source

A uniform flow (\(U_\infty\), \(T_\infty\)) approaches a flat plate of length \(L\) held at a fixed surface temperature. A velocity boundary layer grows from the leading edge, and a thermal boundary layer develops alongside it, governing how effectively heat leaves the surface.

The local coefficient \(h(x)\) is not uniform. It drops through the laminar region as the boundary layer thickens, then rises sharply through the transition zone, and drops more slowly thereafter in the turbulent region.

Interactive demo

Code 
{
  const REC = 5e5;  // canonical critical Reynolds number

  // ── Fluid presets (properties at 25°C, 1 atm) ────────────────────────────
  const presets = {
    Air:     { k: 0.02551, lognu: Math.log10(1.562e-5), Pr: 0.7296 },
    Water:   { k: 0.6065,  lognu: Math.log10(8.91e-7),  Pr: 6.14   },
    Mercury: { k: 8.54,    lognu: Math.log10(1.13e-7),  Pr: 0.0248 },
  };

  // ── Property sliders ─────────────────────────────────────────────────────
  const k_in     = Inputs.range([0.01, 10],   { value: presets.Air.k,
                                                 step: 0.001, label: "k_f (W/m·K)" });
  const lognu_in = Inputs.range([-7.5, -3.5], { value: presets.Air.lognu,
                                                 step: 0.05,
                                                 label: "ν (m²/s)",
                                                 format: v => Math.pow(10, v).toExponential(2) });
  const Pr_in    = Inputs.range([0.005, 10],  { value: presets.Air.Pr,
                                                 step: 0.005, label: "Pr" });

  // ── Physical sliders ──────────────────────────────────────────────────────
  const U_in = Inputs.range([0.5, 20],  { value: 5,   step: 0.5, label: "U∞ (m/s)" });
  const L_in = Inputs.range([0.1, 3.0], { value: 1.0, step: 0.1, label: "L (m)" });

  // ── Preset buttons ────────────────────────────────────────────────────────
  const btnRow = document.createElement("div");
  btnRow.style.marginBottom = "14px";

  const btns = {};
  for (const name of ["Air", "Water", "Mercury"]) {
    const b = document.createElement("button");
    b.textContent = name;
    b.style.cssText = [
      "padding:5px 16px", "margin-right:6px", "border:1.5px solid #aaa",
      "border-radius:4px", "cursor:pointer",
      "font-family:Georgia,serif", "font-size:12px",
      "background:#fff", "transition:background .15s,color .15s",

    btns[name] = b;
    btnRow.append(b);

  function setActive(name) {
    for (const [n, b] of Object.entries(btns)) {
      b.style.background = n === name ? "#4e79a7" : "#fff";
      b.style.color      = n === name ? "#fff"    : "#000";
      b.style.borderColor = n === name ? "#4e79a7" : "#aaa";
    }
  }

  function setSlider(el, val) {
    const inner = el.querySelector("input[type=range]");
    if (inner) {
      inner.value = val;
      inner.dispatchEvent(new Event("input", { bubbles: true }));
    }
  }

  function snap(name) {
    const p = presets[name];
    setSlider(k_in,     p.k);
    setSlider(lognu_in, p.lognu);
    setSlider(Pr_in,    p.Pr);
    setActive(name);
  }

  for (const [name, b] of Object.entries(btns))
    b.addEventListener("click", () => snap(name));
  setActive("Air");

  // ── Plot placeholders ─────────────────────────────────────────────────────
  const pBL = document.createElement("div");
  const ph  = document.createElement("div");

  // ── Render ────────────────────────────────────────────────────────────────
  function render() {
    const k   = k_in.value;
    const nu  = Math.pow(10, lognu_in.value);
    const Pr  = Pr_in.value;
    const U   = U_in.value;
    const L   = L_in.value;

    const xc   = REC * nu / U;
    const Re_L = U * L / nu;

    const N  = 500;
    const xs = Array.from({length: N}, (_, i) => (i + 1) / N * L);

    // Sigmoid blend across the transition zone. Width is schematic —
    // ~5% of Re_c represents a moderately clean flow; real transition width
    // depends on freestream turbulence, roughness, and pressure gradient.
    const W = REC * 0.05;
    const sigma = Re_x => 1 / (1 + Math.exp(-(Re_x - REC) / W));

    // ── Boundary layer thickness ──────────────────────────────────────────
    // δ_lam  = 5x / √Re_x
    // δ_turb = 0.37x / Re_x^0.2  (virtual-origin correction omitted — schematic)
    // δ_T ≈ δ · Pr^(−1/3)
    const blRows = xs.map(x => {
      const Re_x      = U * x / nu;
      const w         = sigma(Re_x);
      const delta_lam  = 5 * x / Math.sqrt(Re_x);
      const delta_turb = 0.37 * x / Math.pow(Re_x, 0.2);
      const delta      = (1 - w) * delta_lam + w * delta_turb;
      return {
        x,
        delta:   delta * 1000,
        delta_T: delta * Math.pow(Pr, -1/3) * 1000,
        regime:  Re_x <= REC ? "lam" : "turb",
      };
    });

    const lamBL  = blRows.filter(r => r.regime === "lam");
    const turbBL = blRows.filter(r => r.regime === "turb");

    // ── h(x) ─────────────────────────────────────────────────────────────
    const hRows = xs.map(x => {
      const Re_x    = U * x / nu;
      const w       = sigma(Re_x);
      const Nu_lam  = 0.332  * Math.pow(Re_x, 0.5) * Math.pow(Pr, 1/3);
      const Nu_turb = 0.0296 * Math.pow(Re_x, 0.8) * Math.pow(Pr, 1/3);
      return { x, h: ((1 - w) * Nu_lam + w * Nu_turb) * k / x };
    });

    // ── Average h ────────────────────────────────────────────────────────
    // Mixed: (k/L)·[0.664·Re_c^½ + 0.037·(Re_L^⁴/⁵ − Re_c^⁴/⁵)]·Pr^⅓
    // Fully laminar (xc ≥ L): (k/L)·0.664·Re_L^½·Pr^⅓
    const A = (xc < L)
      ? 0.664 * Math.pow(REC, 0.5) + 0.037 * (Math.pow(Re_L, 0.8) - Math.pow(REC, 0.8))
      : 0.664 * Math.pow(Re_L, 0.5);
    const h_avg = (k / L) * A * Math.pow(Pr, 1/3);

    const maxDelta = Math.max(...blRows.map(r => Math.max(r.delta, r.delta_T))) * 1.5;
    const maxH     = Math.max(...hRows.map(r => r.h)) * 1.15;
    const xcMark   = xc < L ? [xc] : [];

    // ── BL plot ───────────────────────────────────────────────────────────
    const plotBL = Plot.plot({
      width: 680, height: 190,
      marginLeft: 52, marginBottom: 36, marginTop: 16,
      style: { fontFamily: "Georgia, serif", fontSize: "11px" },
      x: { label: "x (m) →", domain: [0, L] },
      y: { label: "↑ δ (mm)", domain: [0, maxDelta] },
      marks: [
        Plot.gridX({ stroke: "#e0e0e0" }),
        Plot.gridY({ stroke: "#e0e0e0" }),
        Plot.areaY(lamBL,  { x: "x", y: "delta",   fill: "#4e79a7", fillOpacity: 0.18 }),
        Plot.line(lamBL,   { x: "x", y: "delta",   stroke: "#4e79a7", strokeWidth: 2 }),
        Plot.areaY(turbBL, { x: "x", y: "delta",   fill: "#f28e2b", fillOpacity: 0.18 }),
        Plot.line(turbBL,  { x: "x", y: "delta",   stroke: "#f28e2b", strokeWidth: 2 }),
        Plot.line(lamBL,   { x: "x", y: "delta_T", stroke: "#4e79a7",
                             strokeDasharray: "5,3", strokeWidth: 1.5 }),
        Plot.line(turbBL,  { x: "x", y: "delta_T", stroke: "#f28e2b",
                             strokeDasharray: "5,3", strokeWidth: 1.5 }),
        Plot.ruleY([0], { stroke: "#333", strokeWidth: 3 }),
        Plot.ruleX(xcMark, { stroke: "#e15759", strokeWidth: 1.5, strokeDasharray: "6,3" }),
      ]
    });

    // ── h(x) plot ─────────────────────────────────────────────────────────
    const ploth = Plot.plot({
      width: 680, height: 250,
      marginLeft: 62, marginBottom: 36, marginTop: 10,
      style: { fontFamily: "Georgia, serif", fontSize: "11px" },
      x: { label: "x (m) →", domain: [0, L] },
      y: { label: "↑ h (W/m²·K)", domain: [0, maxH] },
      marks: [
        Plot.gridX({ stroke: "#e0e0e0" }),
        Plot.gridY({ stroke: "#e0e0e0" }),
        Plot.line(hRows, { x: "x", y: "h", stroke: "#444", strokeWidth: 2 }),
        Plot.ruleX(xcMark, { stroke: "#e15759", strokeWidth: 1.5, strokeDasharray: "6,3" }),
        // h̄ — short horizontal segment at x = L (evoking the overline notation)
        Plot.link([{ x1: L * 0.93, y1: h_avg, x2: L, y2: h_avg }], {

          stroke: "#59a14f", strokeWidth: 2.5,

What to look for

Notice the axis units. The \(x\)-axis is in meters, but the \(y\)-axis in the boundary layer plot is in millimeters — boundary layers are very thin! A plate a meter long might have a laminar BL only a few millimeters thick at its trailing edge.

h drops through the laminar region — as \(x^{-1/2}\). The boundary layer thickens away from the leading edge, the temperature gradient at the wall decreases, and so does the local heat flux for a fixed \(\Delta T\). This is counterintuitive at first: moving further along the plate makes heat transfer worse, locally.

h rises sharply through the transition zone. As the boundary layer goes turbulent near \(x_c = \text{Re}_c \cdot \nu / U_\infty\), turbulent mixing transports heat perpendicular to the plate far more effectively, and \(h\) rises rapidly. (Drag increases too — the same mixing that improves heat transfer also raises skin friction.) The smooth S-curve in the demo is schematic: real transition width depends on freestream turbulence intensity, surface roughness, and pressure gradient — it can be very sharp in a clean wind tunnel or spread over 30–40% of \(x_c\) in a noisy environment.

h drops more slowly in the turbulent region — as \(x^{-1/5}\). The turbulent BL still thickens, but the radial mixing partially compensates.

Things to try: - Increase \(U_\infty\) until the transition marker appears — watch \(\bar{h}\) jump as turbulence takes over more of the plate. - Switch to Water — \(\bar{h}\) increases dramatically. Water’s high \(k\) and \(c_p\) make it a far more effective convective coolant. - Try Mercury — \(\text{Pr} \approx 0.025\), so the thermal boundary layer is much thicker than the velocity BL. The \(h\) values are very high (high \(k_f\)) but the Pr scaling pulls them back. A genuinely different regime. - Adjust individual properties after snapping a preset — e.g., what if you kept air’s \(\nu\) but gave it water’s \(k\)?

Worked example — mixed boundary layer

Air at \(T_\infty = 25\,^\circ\text{C}\), \(U_\infty = 10\) m/s flows over a plate \(L = 2\) m long. Find \(\bar{h}\).

Properties (25°C, matching the demo defaults): \[k_f = 0.02551\;\text{W/m·K}, \quad \nu = 1.562\times10^{-5}\;\text{m}^2/\text{s}, \quad \text{Pr} = 0.730\]

Step 1 — Check whether transition occurs on the plate.

\[\text{Re}_L = \frac{U_\infty L}{\nu} = \frac{10 \times 2}{1.562\times10^{-5}} = 1.28\times10^6 > \text{Re}_c = 5\times10^5 \quad \checkmark \text{ mixed}\]

Transition location:

\[x_c = \frac{\text{Re}_c \cdot \nu}{U_\infty} = \frac{5\times10^5 \times 1.562\times10^{-5}}{10} = 0.78\;\text{m}\]

So the first 78% of the plate is laminar, the rest is turbulent.

Step 2 — Apply the mixed average correlation.

Integrating the local laminar correlation over \([0, x_c]\) and the turbulent one over \([x_c, L]\):

\[\overline{h} = \frac{k_f}{L}\left[0.664\,\text{Re}_c^{1/2} + 0.037\!\left(\text{Re}_L^{4/5} - \text{Re}_c^{4/5}\right)\right]\text{Pr}^{1/3}\]

Plugging in:

\[= \frac{0.02551}{2}\!\left[0.664\sqrt{5\times10^5} + 0.037\!\left((1.28\times10^6)^{0.8} - (5\times10^5)^{0.8}\right)\right](0.730)^{1/3}\]

\[= 0.01276\!\left[\underbrace{0.664\times707}_{469} + 0.037\!\underbrace{\left(76{,}850 - 36{,}200\right)}_{40{,}650}\right]\times 0.900\]

\[= 0.01276\times\left[469 + 1504\right]\times 0.900 = 0.01276\times1973\times0.900\]

\[\boxed{\bar{h} \approx 22.6\;\text{W/m}^2\text{K}}\]

Set \(U_\infty = 10\), \(L = 2\) in the demo (Air preset) and confirm the green dot reads approximately 22.6. Now compare to a fully laminar plate of the same length: set \(U_\infty = 3\) m/s (so \(\text{Re}_L = 3.84\times10^5 < 5\times10^5\)). The laminar-only \(\bar{h}\) drops to roughly 8 W/m²·K — nearly a factor of three lower, despite the fully turbulent plate running at more than three times the speed.

## What to look for {#sec-observations}

**Notice the axis units.** The $x$-axis is in meters, but the $y$-axis in the boundary layer plot is in *millimeters* — boundary layers are very thin! A plate a meter long might have a laminar BL only a few millimeters thick at its trailing edge.

**h drops through the laminar region** — as $x^{-1/2}$. The boundary layer thickens away from the leading edge, the temperature gradient at the wall decreases, and so does the local heat flux for a fixed $\Delta T$. This is counterintuitive at first: moving further along the plate makes heat transfer *worse*, locally.

**h rises sharply through the transition zone.** As the boundary layer goes turbulent near $x_c = \text{Re}_c \cdot \nu / U_\infty$, turbulent mixing transports heat perpendicular to the plate far more effectively, and $h$ rises rapidly. (Drag increases too — the same mixing that improves heat transfer also raises skin friction.) The smooth S-curve in the demo is schematic: real transition width depends on freestream turbulence intensity, surface roughness, and pressure gradient — it can be very sharp in a clean wind tunnel or spread over 30–40% of $x_c$ in a noisy environment.

**h drops more slowly in the turbulent region** — as $x^{-1/5}$. The turbulent BL still thickens, but the radial mixing partially compensates.

**Things to try:**
- *Increase $U_\infty$* until the transition marker appears — watch $\bar{h}$ jump as turbulence takes over more of the plate.
- *Switch to Water* — $\bar{h}$ increases dramatically. Water's high $k$ and $c_p$ make it a far more effective convective coolant.
- *Try Mercury* — $\text{Pr} \approx 0.025$, so the thermal boundary layer is much *thicker* than the velocity BL. The $h$ values are very high (high $k_f$) but the Pr scaling pulls them back. A genuinely different regime.
- *Adjust individual properties* after snapping a preset — e.g., what if you kept air's $\nu$ but gave it water's $k$?

## Worked example — mixed boundary layer {#sec-worked}

Air at $T_\infty = 25\,^\circ\text{C}$, $U_\infty = 10$ m/s flows over a plate $L = 2$ m long. Find $\bar{h}$.

**Properties** (25°C, matching the demo defaults):
$$k_f = 0.02551\;\text{W/m·K}, \quad \nu = 1.562\times10^{-5}\;\text{m}^2/\text{s}, \quad \text{Pr} = 0.730$$

**Step 1 — Check whether transition occurs on the plate.**

$$\text{Re}_L = \frac{U_\infty L}{\nu} = \frac{10 \times 2}{1.562\times10^{-5}} = 1.28\times10^6 > \text{Re}_c = 5\times10^5 \quad \checkmark \text{ mixed}$$

Transition location:

$$x_c = \frac{\text{Re}_c \cdot \nu}{U_\infty} = \frac{5\times10^5 \times 1.562\times10^{-5}}{10} = 0.78\;\text{m}$$

So the first 78% of the plate is laminar, the rest is turbulent.

**Step 2 — Apply the mixed average correlation.**

Integrating the local laminar correlation over $[0, x_c]$ and the turbulent one over $[x_c, L]$:

$$\overline{h} = \frac{k_f}{L}\left[0.664\,\text{Re}_c^{1/2} + 0.037\!\left(\text{Re}_L^{4/5} - \text{Re}_c^{4/5}\right)\right]\text{Pr}^{1/3}$$

Plugging in:

$$= \frac{0.02551}{2}\!\left[0.664\sqrt{5\times10^5} + 0.037\!\left((1.28\times10^6)^{0.8} - (5\times10^5)^{0.8}\right)\right](0.730)^{1/3}$$

$$= 0.01276\!\left[\underbrace{0.664\times707}_{469} + 0.037\!\underbrace{\left(76{,}850 - 36{,}200\right)}_{40{,}650}\right]\times 0.900$$

$$= 0.01276\times\left[469 + 1504\right]\times 0.900 = 0.01276\times1973\times0.900$$

$$\boxed{\bar{h} \approx 22.6\;\text{W/m}^2\text{K}}$$

Set $U_\infty = 10$, $L = 2$ in the demo (Air preset) and confirm the green dot reads approximately 22.6. Now compare to a fully laminar plate of the same length: set $U_\infty = 3$ m/s (so $\text{Re}_L = 3.84\times10^5 < 5\times10^5$). The laminar-only $\bar{h}$ drops to roughly 8 W/m²·K — nearly a factor of three lower, despite the fully turbulent plate running at more than three times the speed.
